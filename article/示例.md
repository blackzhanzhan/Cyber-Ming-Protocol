# ============== H3 图谱裂变 ==============
def build_cypher_batch(
    lecture_title: str,
    category: str,
    points: list[dict],
    bv_id: str = DEFAULT_BV_ID,
    relations: list[dict] | None = None,   # ← v6新增：接收多元关系
) -> str:
    """四层图谱：Lecture + Category + KnowledgePoint + 多元Relations"""
    lines = []

    lines.append(f'MERGE (L:Lecture {{title: "{lecture_title}", bv_id: "{bv_id}"}})')
    lines.append(f'MERGE (C:Category {{name: "{category}"}})')
    lines.append(f"MERGE (L)-[:BELONGS_TO]->(C)")

    point_names = []
    for i, pt in enumerate(points):
        s = pt.get("start_ms", 0)
        e = pt.get("end_ms", 0)
        pt_name = pt.get("point_name", "unknown")[:20]
        point_names.append(pt_name)

        lines.append(
            f'MERGE (P{i}:KnowledgePoint {{name: "{pt_name}", start_ms: {s}, end_ms: {e}, lecture: "{lecture_title}"}})'
        )

    # ← v6核心：遍历relations生成多元关系边
    rels = relations or []
    for rel in rels:
        source = str(rel.get("source_title", "")).strip()[:20]
        target = str(rel.get("target_title", "")).strip()[:20]
        rel_type = str(rel.get("relation_type", "")).strip().upper()
        if source and target and rel_type:
            source_idx = next((j for j, name in enumerate(point_names) if name == source), None)
            target_idx = next((j for j, name in enumerate(point_names) if name == target), None)
            if source_idx is not None and target_idx is not None:
                lines.append(f"MERGE (P{source_idx})-[r:{rel_type}]->(P{target_idx})")  # ← 星辰连线！
            elif source and target:
                lines.append(f'MERGE (S:Entity {{title: "{source}"}})-[r:{rel_type}]->(T:Entity {{title: "{target}"}})')

    # 保留CONTAINS作为底座
    for i, pt in enumerate(points):
        lines.append(f"MERGE (L)-[:CONTAINS]->(P{i})")

    return "\n".join(lines)

```python
def build_cypher_batch(
    lecture_title: str,
    category: str,
    points: list[dict],
    bv_id: str = DEFAULT_BV_ID,
    relations: list[dict] | None = None,
) -> str:
    """四层图谱：Lecture + Category + KnowledgePoint + 多元Relations
    所有变量名使用 lecture_title 派生前缀，确保全局唯一
    """
    lines = []

    # ========== 1. Lecture + Category（唯一前缀）==========
    lecture_prefix = lecture_title.replace("-", "_").replace(" ", "_")[:8]
    lecture_var = f"L_{lecture_prefix}"              # 例: L_p08_two_
    category_var = f"C_{lecture_prefix}"             # 例: C_p08_two_
    
    lines.append(f'MERGE ({lecture_var}:Lecture {{title: "{lecture_title}", bv_id: "{bv_id}"}})')
    lines.append(f'MERGE ({category_var}:Category {{name: "{category}"}})')
    lines.append(f"MERGE ({lecture_var})-[:BELONGS_TO]->({category_var})")

    # ========== 2. KnowledgePoint 节点（唯一前缀+索引）==========
    point_names = []
    point_vars = []
    for i, pt in enumerate(points):
        s = pt.get("start_ms", 0)
        e = pt.get("end_ms", 0)
        pt_name = pt.get("point_name", "unknown")[:20]
        point_names.append(pt_name)
        point_var = f"{lecture_prefix}P{i}"          # 例: p08_two_P0, p08_two_P1
        point_vars.append(point_var)
        lines.append(
            f'MERGE ({point_var}:KnowledgePoint {{name: "{pt_name}", start_ms: {s}, end_ms: {e}, lecture: "{lecture_title}"}})'
        )

    # ========== 3. 关系边（全局递增计数器）==========
    rels = relations or []
    seen_relations = set()
    rel_counter = 0                                     # 全局递增计数器
    for rel in rels:
        source = str(rel.get("source_title", "")).strip()[:20]
        target = str(rel.get("target_title", "")).strip()[:20]
        rel_type = str(rel.get("relation_type", "")).strip().upper()
        
        if not (source and target and rel_type):
            continue
            
        # 查找知识点索引
        source_idx = next((j for j, name in enumerate(point_names) if name == source), None)
        target_idx = next((j for j, name in enumerate(point_names) if name == target), None)
        
        if source_idx is None or target_idx is None:
            continue
            
        # 去重检查
        rel_key = (source, target, rel_type)
        if rel_key in seen_relations:
            continue
        seen_relations.add(rel_key)
        
        # 计算时空属性（两知识点时间的中值区间）
        s_ms = points[source_idx].get("start_ms", 0)
        e_ms = points[source_idx].get("end_ms", 0)
        t_ms = points[target_idx].get("start_ms", 0)
        t_em = points[target_idx].get("end_ms", 0)
        start_ms = max(s_ms, t_ms)
        end_ms = min(e_ms, t_em)
        if end_ms <= start_ms:
            start_ms = (s_ms + t_ms) // 2
            end_ms = (e_ms + t_em) // 2
        
        # 生成唯一关系变量名
        rel_counter += 1
        rel_var = f"r{rel_counter}"                    # 例: r1, r2, r3（全局递增）
        
        lines.append(
            f"MERGE ({point_vars[source_idx]})-[{rel_var}:{rel_type} {{start_ms: {start_ms}, end_ms: {end_ms}}}]->({point_vars[target_idx]})"
        )

    # ========== 4. CONTAINS 边（使用已定义的唯一变量名）==========
    for i, pt in enumerate(points):
        lines.append(f"MERGE ({lecture_var})-[:CONTAINS]->({point_vars[i]})")

    return "\n".join(lines)
```

| 变量类型 | 命名格式 | 示例 |
|---|---|---|
| Lecture | `L_{lecture_prefix}` | `L_p08_two_` |
| Category | `C_{lecture_prefix}` | `C_p08_two_` |
| KnowledgePoint | `{lecture_prefix}P{index}` | `p08_twoP0`, `p08_twoP1` |

## 《关系变量全局防冲突方案策论》

### 1. 问题根因分析

| 现状 | 失效原因 |
|------|----------|
| `rel_counter = len(seen_relations)` | 只在**单文件内**从1计数，28个lecture合并后 r1/r2/r3 重复撞车 |
| 每个lecture独立生成 | 变量名作用域是**整个Cypher文件**，非单条MERGE |

---

### 2. 方案对比

| 方案 | 核心思路 | 代码侵入性 | Neo4j效率 | 变量示例 |
|------|----------|------------|-----------|----------|
| **A: 讲次前缀拼接** | `rel_counter` 改为 `(lecture_idx * 100) + local_counter` | 低 | ⭐⭐⭐⭐⭐ | `r301`, `r105` |
| **B: 省略变量名** | 直接 `[r:TYPE]` 或 `[:TYPE]`，不指定变量名 | 最低 | ⭐⭐⭐⭐⭐ | `[:PREREQUISITE]` |
| **C: MD5哈希** | `hash(source+target+type)[:6]` 生成唯一变量名 | 中 | ⭐⭐⭐ | `r7a3b2` |
| **D: UUID** | 每条关系用 `r{uuid4()[:8]}` | 中 | ⭐⭐⭐ | `r1a2b3c4d` |

---

### 3. 优劣势详析

| 方案 | 优点 | 缺点 |
|------|------|------|
| **A: 讲次前缀** | 直观、可调试、变量名有规律 | 需传入 lecture 索引 |
| **B: 省略变量名** | 无需任何代码改动，Neo4j自动分配 | 无法通过变量名引用关系 |
| **C: MD5哈希** | 完全去中心化 | 变量名无意义、难调试 |
| **D: UUID** | 绝对唯一 | 变量名太长、不可读 |

---

### 4. 臣的推荐

**首选方案B（省略变量名）**，理由：
- 零代码改动
- Cypher中变量名大多无实际用途
- Neo4j内部自动处理，性能最优

**备选方案A（讲次前缀）**，若需要通过变量名引用关系时使用。
